## Core Types

### OAuthConfig

OAuth authentication configuration.

```typescript
interface OAuthConfig {
  clientId: string;
  clientSecret?: string;
  redirectUri: string;
  scope?: 'team' | 'project';
  tokenExchangeUrl?: string;
}
```

<ResponseField name="clientId" type="string" required>
  OAuth client ID.
</ResponseField>

<ResponseField name="clientSecret" type="string">
  OAuth client secret (optional for public clients).
</ResponseField>

<ResponseField name="redirectUri" type="string" required>
  OAuth redirect URI after authentication.
</ResponseField>

<ResponseField name="scope" type="'team' | 'project'">
  OAuth scope - either 'team' or 'project' level access.
</ResponseField>

<ResponseField name="tokenExchangeUrl" type="string">
  Custom URL for token exchange endpoint.
</ResponseField>

## Log Types

### LogEntry

Structure for a log entry.

```typescript
interface LogEntry {
  timestamp: number;
  topic: string;
  function?: {
    type?: string;
    path?: string;
    cached?: boolean;
    request_id?: string;
  };
  log_level?: string;
  message?: string;
  execution_time_ms?: number;
  status?: string;
  error_message?: string;
  usage?: {
    database_read_bytes?: number;
    database_write_bytes?: number;
    file_storage_read_bytes?: number;
    file_storage_write_bytes?: number;
    vector_storage_read_bytes?: number;
    vector_storage_write_bytes?: number;
    action_memory_used_mb?: number;
  };
  system_code?: string;
  audit_log_action?: string;
  audit_log_metadata?: string;
  raw: any;
}
```

<ResponseField name="timestamp" type="number" required>
  Unix timestamp of the log entry.
</ResponseField>

<ResponseField name="topic" type="string" required>
  Log topic/category.
</ResponseField>

<ResponseField name="function" type="object">
  Information about the function that generated this log.
  
  <Expandable title="properties">
    <ResponseField name="type" type="string">
      Function type (query, mutation, action).
    </ResponseField>
    
    <ResponseField name="path" type="string">
      Function path/identifier.
    </ResponseField>
    
    <ResponseField name="cached" type="boolean">
      Whether the result was cached.
    </ResponseField>
    
    <ResponseField name="request_id" type="string">
      Request ID for correlation.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="execution_time_ms" type="number">
  Function execution time in milliseconds.
</ResponseField>

<ResponseField name="usage" type="object">
  Resource usage statistics.
  
  <Expandable title="properties">
    <ResponseField name="database_read_bytes" type="number">
      Bytes read from database.
    </ResponseField>
    
    <ResponseField name="database_write_bytes" type="number">
      Bytes written to database.
    </ResponseField>
    
    <ResponseField name="file_storage_read_bytes" type="number">
      Bytes read from file storage.
    </ResponseField>
    
    <ResponseField name="file_storage_write_bytes" type="number">

    </ResponseField>
    
    <ResponseField name="action_memory_used_mb" type="number">
      Memory used by action in MB.
    </ResponseField>
  </Expandable>
</ResponseField>

## Table Types

### TableField

Schema field definition for a table.

```typescript
interface TableField {
  fieldName: string;
  optional: boolean;
  shape: {
    type: string;
    fields?: TableField[];
    tableName?: string;
    value?: any;
    float64Range?: {
      hasSpecialValues: boolean;
    };
    shape?: {
      type: string;
      tableName?: string;
      value?: any;
      fields?: TableField[];
    };
  };
}
```

### TableSchema

Complete schema for a table.

```typescript
interface TableSchema {
  type: string;
  fields: TableField[];
}
```

### TableDefinition

Map of table names to their schemas.

```typescript
interface TableDefinition {
  [key: string]: TableSchema;
}
```

### TableDocument

Document stored in a table.

```typescript
interface TableDocument {

}
```

<ResponseField name="_id" type="string" required>

<ResponseField name="_initialEditMode" type="boolean">

## Filter Types

### FilterClause

Single filter condition.

```typescript
interface FilterClause {
  field: string;
  op: 'eq' | 'neq' | 'gt' | 'gte' | 'lt' | 'lte' | 'isType' | 'isNotType';
  value: any;
  enabled: boolean;
}
```

<ResponseField name="field" type="string" required>
  Field name to filter on.
</ResponseField>

<ResponseField name="op" type="FilterOperation" required>
  Filter operation to apply.
  
  Options:
  - `eq` - Equals
  - `neq` - Not equals
  - `gt` - Greater than
  - `gte` - Greater than or equal
  - `lt` - Less than
  - `lte` - Less than or equal
  - `isType` - Is of type
  - `isNotType` - Is not of type
</ResponseField>

<ResponseField name="value" type="any" required>
  Value to compare against.
</ResponseField>

<ResponseField name="enabled" type="boolean" required>
  Whether this filter is currently active.
</ResponseField>

### FilterExpression

Combination of multiple filter clauses.

```typescript
interface FilterExpression {
  clauses: FilterClause[];
}
```
